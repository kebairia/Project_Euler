* Problem 2 - Even Fibonacci numbers
#+begin_src python :results output :exports both
  is_even = lambda x: x % 2 == 0
  def fib(limit):
      ans = 1 
      lst = []
      x, y = 0, 1
      while x <= limit:
          lst.append(ans)
          ans += y
          x, y = y, x + y
      return lst
  print("The result is: ", sum([x for x in fib(4000000) if is_even(x)]))
#+end_src
